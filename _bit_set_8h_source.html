268inline void bit_set_fill_in_range_begin_end(BitSet* const bit_set, const bool value, const uint64_t begin, const uint64_t end)
269{
270 uint8_t to_add = 1, to_sub = 1;
271 // create begin_chunk and fill the first byte with it
272 if (begin % 8)
273 {
274 const uint16_t end_bit = (begin / 8 == end / 8) ? end % 8 : 8;
275 if (value)
276 {
277 for (uint16_t i = begin % 8; i < end_bit; ++i)
278 *(bit_set->data + begin / 8) |= 1 << i;
279 }
280 else
281 {
282 for (uint16_t i = begin % 8; i < end_bit; ++i)
283 *(bit_set->data + begin / 8) &= ~(1 << i);
284 }
285 }
286 else
287 to_add = 0;
288 if (end % 8 && begin / 8 != end / 8)
289 {
290 if (value)
291 {
292 for (uint16_t i = 0; i < end % 8; ++i)
293 *(bit_set->data + end / 8) |= 1u << i;
294 }
295 else
296 {
297 for (uint16_t i = 0; i < end % 8; ++i)
298 *(bit_set->data + end / 8) &= ~(1u << i);
299 }
300 }
301 else
302 to_sub = 0;
303
304 memset(bit_set->data + begin / 8 + to_add, value ? 255u : 0u, (end - begin) / 8u - to_sub);
305}

738void bit_set_push_back(BitSet* const bit_set, const bool value)
739{
740 if (bit_set->size % 8)
741 {
742 if (value)
743 *(bit_set->data + bit_set->size / 8) |= 1u << bit_set->size % 8;
744 else
745 *(bit_set->data + bit_set->size / 8) &= ~(1u << bit_set->size % 8);
746 }
747 else
748 {
749 uint8_t* new_data = (uint8_t*)malloc((bit_set->storage_size + 1) * sizeof(uint8_t));
750 if (bit_set->data)
751 {
752 memcpy(new_data, bit_set->data, bit_set->storage_size);
753 free(bit_set->data);
754 }
755 bit_set->data = new_data;
756 *(bit_set->data + bit_set->storage_size++) = value ? 1u : 0u;
757 }
758 ++bit_set->size;
759}
