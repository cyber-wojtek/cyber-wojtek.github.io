271inline void bit_set_fill_in_range_begin_end(BitSet* const bit_set, const bool value, const uint64_t begin, const uint64_t end)
272{
273 uint8_t to_add = 1, to_sub = 1;
274 // create begin_chunk and fill the first byte with it
275 if (begin % 8)
276 {
277 const uint16_t end_bit = (begin / 8 == end / 8) ? end % 8 : 8;
278 if (value)
279 {
280 for (uint16_t i = begin % 8; i < end_bit; ++i)
281 *(bit_set->data + begin / 8) |= 1 << i;
282 }
283 else
284 {
285 for (uint16_t i = begin % 8; i < end_bit; ++i)
286 *(bit_set->data + begin / 8) &= ~(1 << i);
287 }
288 }
289 else
290 to_add = 0;
291 if (end % 8 && begin / 8 != end / 8)
292 {
293 if (value)
294 {
295 for (uint16_t i = 0; i < end % 8; ++i)
296 *(bit_set->data + end / 8) |= (uint8_t)1u << i;
297 }
298 else
299 {
300 for (uint16_t i = 0; i < end % 8; ++i)
301 *(bit_set->data + end / 8) &= ~((uint8_t)1u << i);
302 }
303 }
304 else
305 to_sub = 0;
306
307 memset(bit_set->data + begin / 8 + to_add, value ? 255u : 0u, (end - begin) / 8u - to_sub);
308}
