261inline void bit_set_fill_in_range_begin_end(BitSet* const bit_set, const bool value, const uint64_t begin, const uint64_t end)
262{
263 uint8_t to_add = 1, to_sub = 1;
264 // create begin_chunk and fill the first byte with it
265 if (begin % 8)
266 {
267 const uint16_t end_bit = (begin / 8 == end / 8) ? end % 8 : 8;
268 if (value)
269 {
270 for (uint16_t i = begin % 8; i < end_bit; ++i)
271 *(bit_set->data + begin / 8) |= 1 << i;
272 }
273 else
274 {
275 for (uint16_t i = begin % 8; i < end_bit; ++i)
276 *(bit_set->data + begin / 8) &= ~(1 << i);
277 }
278 }
279 else
280 to_add = 0;
281 if (end % 8 && begin / 8 != end / 8)
282 {
283 if (value)
284 {
285 for (uint16_t i = 0; i < end % 8; ++i)
286 *(bit_set->data + end / 8) |= 1u << i;
287 }
288 else
289 {
290 for (uint16_t i = 0; i < end % 8; ++i)
291 *(bit_set->data + end / 8) &= ~(1u << i);
292 }
293 }
294 else
295 to_sub = 0;
296
297 memset(bit_set->data + begin / 8 + to_add, value ? 255u : 0u, (end - begin) / 8u - to_sub);
298}

731void bit_set_push_back(BitSet* const bit_set, const bool value)
732{
733 if (bit_set->size % 8)
734 {
735 if (value)
736 *(bit_set->data + bit_set->size / 8) |= 1u << bit_set->size % 8;
737 else
738 *(bit_set->data + bit_set->size / 8) &= ~(1u << bit_set->size % 8);
739 }
740 else
741 {
742 uint8_t* new_data = (uint8_t*)malloc((bit_set->storage_size + 1) * sizeof(uint8_t));
743 if (bit_set->data)
744 {
745 memcpy(new_data, bit_set->data, bit_set->storage_size);
746 free(bit_set->data);
747 }
748 bit_set->data = new_data;
749 *(bit_set->data + bit_set->storage_size++) = value ? 1u : 0u;
750 }
751 ++bit_set->size;
752}
