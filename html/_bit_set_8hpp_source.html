408 void fillInRangeOptimized(const bool& value, const uint64_t& begin, const uint64_t& end, const uint64_t& step)
409 {
410 // Initialize variables
411 uint64_t chunks_size, current_chunk = begin / chunk_size + 1 + step / chunk_size, current_offset = 0;
412 uint16_t offset;
413 const uint64_t end_chunk = end / chunk_size + (end % chunk_size ? 1 : 0);
414
415 // Determine the size of chunks based on step and chunk size
416 if ((step % 2 || step <= chunk_size) && chunk_size % step) {
417 chunks_size = (std::min)(storage_size, step);
418 }
419 else if (!(chunk_size % step))
420 chunks_size = 1;
421 else if (!(step % chunk_size))
422 chunks_size = step / chunk_size;
423 else
424 {
425 // GCD of step and chunk_size
426 if (step % chunk_size)
427 {
428 uint64_t a = step, b = chunk_size, t = chunk_size;
429 while (b) {
430 t = b;
431 b = a % b;
432 a = t;
433 }
434 chunks_size = a;
435 }
436 else
437 chunks_size = 1;
438 }
439
440 // Calculate the offset
441 if (begin < chunk_size)
442 {
443 offset = (chunk_size - begin) % step;
444 if (offset)
445 offset = step - offset;
446 }
447 else
448 {
449 offset = (begin - chunk_size) % step;
450 }
451
452 if (offset)
453 offset = ((chunk_size - offset) + step / chunk_size * chunk_size) % step;
454
455 std::cout << "offset: " << offset << '\n';
456
457 // Create and apply the beginning chunk
458 {
459 const uint16_t end_bit = (begin / chunk_size == end / chunk_size) ? end % chunk_size : chunk_size;
460 if (value)
461 {
462 for (uint16_t i = begin % chunk_size; i < end_bit; i += step)
463 *(data + begin / chunk_size) |= static_cast<T>(1) << i;
464 }
465 else
466 {
467 for (uint16_t i = begin % chunk_size; i < end_bit; i += step)
468 *(data + begin / chunk_size) &= ~(static_cast<T>(1) << i);
469 }
470 }
471
472
473 // Fill with appropriate chunk
474 std::cout << chunks_size << " chunks\n";
475 std::cout << (std::min)(chunks_size + begin / chunk_size, storage_size) << " chunks\n";
476 for (uint64_t i = 0; i < (std::min)(chunks_size, storage_size); ++i)
477 {
478 // Generate chunk for the current iteration
479 T chunk = 0;
480
481 if (value)
482 {
483 for (uint16_t j = !i ? offset : 0; j < chunk_size; ++j)
484 {
485 std::cout << current_chunk * chunk_size + j - offset << '\n';
486 if (!((current_chunk * chunk_size + j - offset) % step))
487 chunk |= static_cast<T>(1) << j;
488 }
489 }
490 else
491 {
492 chunk = (std::numeric_limits<T>::max)();
493 for (uint16_t j = (!i ? offset : 0); j < chunk_size; ++j)
494 {
495 if (!((current_chunk * chunk_size + j - offset) % step))
496 chunk &= ~(static_cast<T>(1) << j);
497 }
498 }
499
500 // print the chunk
501 for (uint16_t j = 0; j < chunk_size; ++j)
502 {
503 std::cout << ((chunk & static_cast<T>(1) << j) >> j);
504 }
505
506 std::cout << '\n';
507
508 // Apply the chunk
509 for (uint64_t j = current_chunk; j < storage_size; ++j)
510 {
511 if (j == end_chunk - 1 && end % chunk_size)
512 {
513 // Remove bits that overflow the range
514 if (value)
515 {
516 for (uint16_t k = end % chunk_size; k < chunk_size; ++k)
517 chunk &= ~(static_cast<T>(1) << k);
518 *(data + j) |= chunk;
519 }
520 else
521 {
522 for (uint16_t k = end % chunk_size; k < chunk_size; ++k)
523 chunk |= static_cast<T>(1) << k;
524 *(data + j) &= chunk;
525 }
526 break;
527 }
528 if (value)
529 *(data + j) |= chunk;
530 else
531 *(data + j) &= chunk;
532 }
533 ++current_chunk;
534 }
